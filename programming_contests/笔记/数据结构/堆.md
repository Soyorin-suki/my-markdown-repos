# 堆

常用STL: `priority_queue`实现
code:
```c++
struct heap{
	vector<ll>ele;
	ll n=0;
	ll f=0;
	heap():ele(1){}
	bool cmp(ll a,ll b){
		if(f==1)return a<b;
		else return a>b;
	}
	void up(ll x){
		while(x>1&&cmp(ele[x],ele[x>>1])){
			swap(ele[x],ele[x>>1]);
			x>>=1;
		}
	}
	void down(ll x){
		while(x*2<=n){
			ll t=x*2;
			if(t+1<=n&&cmp(ele[t+1],ele[t]))t++;
			if(!cmp(ele[t],ele[x]))break;
			swap(ele[x],ele[t]);
			x=t;
		}
	}
	void push(ll x){
		ele.push_back(x);
		n++;
		up(n);
	}
	ll top(void){
		return ele[1];
	}
	void pop(void){
		swap(ele[1],ele[n]);
		n--;
		ele.pop_back();
		down(1);
	}
};
```
