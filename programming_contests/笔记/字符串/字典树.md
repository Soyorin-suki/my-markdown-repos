struct node{
	ll nxt[26];
}trie[N];
ll vis[N];
ll tot;
ll n,m;
void insert(string&s,ll id){
	ll now=0;
	for(ll i=0;i<s.length();++i){
		ll ch=s[i]-'a';
		if(!trie[now].nxt[ch]){
			trie[now].nxt[ch]=++tot;
		}
		now=trie[now].nxt[ch];
	}
	vis[now]=id;
}
ll search(string&s){
	ll now=0;
	for(ll i=0;i<s.length();++i){
		ll ch=s[i]-'a';
		if(!trie[now].nxt[ch]){
			return 0;
		}
		now=trie[now].nxt[ch];
	}
	if(vis[now])return vis[now]++;
	else 0;
}
